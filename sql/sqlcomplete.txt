unique key;
.unique key only 1 null value take
.we can have multiple unique key in table but only one primary key

mysql> create table demo(
    -> id int ,
    -> firstname varchar(20) not null,
    -> lastname varchar(10) not null,
    -> age int not null,
    -> unique key(id)
    -> );
